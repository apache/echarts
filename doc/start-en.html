<!DOCTYPE html>
<head>
    <meta charset="utf-8">
    <title>ECharts</title>

            paths:{

            }
        });
    </script>
</body>
4、Dynamic loading echarts and required chart in <script>,callback function to initialize the chart and drive generates charts,option to see API & Doc 
<!DOCTYPE html>
<head>
    <meta charset="utf-8">
    <title>ECharts</title>
    <!-- From Baidu CDN -->
    <script src="http://s1.bdstatic.com/r/www/cache/ecom/esl/1-6-10/esl.js"></script>
</head>
<body>
    <!-- Prepare a Dom with size(width and height) for ECharts -->
    <div id="main" style="height:400px"></div>
    <script type="text/javascript">
        // Path configuration
        require.config({
            paths:{ 
                'echarts' : 'http://echarts.baidu.com/build/echarts',
                'echarts/chart/bar' : 'http://echarts.baidu.com/build/echarts'
            }
        });
        
        // Use
        require(
            [
                'echarts',
                'echarts/chart/bar' // Using the bar will load the bar module,loaded on demand
            ],
            function (ec) {
                // Based on the prepared dom,initialization echarts chart
                var myChart = ec.init(document.getElementById('main')); 
                
                var option = {
                    tooltip: {
                        show: true
                    },
                    legend: {
                        data:['Sales volume']
                    },
                    xAxis : [
                        {
                            type : 'category',
                            data : ["Shirt","Cardigan","Chiffon shirt","Pants","High-heeled shoes","Socks"]
                        }
                    ],
                    yAxis : [
                        {
                            type : 'value'
                        }
                    ],
                    series : [
                        {
                            "name":"Sales volume",
                            "type":"bar",
                            "data":[5, 20, 40, 10, 10, 20]
                        }
                    ]
                };
        
                // for Echarts object loading data
                myChart.setOption(option); 
            }
        );
    </script>
</body>
